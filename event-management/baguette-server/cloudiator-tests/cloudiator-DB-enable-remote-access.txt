********************************************************************
* INSTRUCTION to enable remote access to cloudiator mysql database *
********************************************************************

sudo vi /etc/mysql/mysql.conf.d/mysqld.cnf
....and comment out the line:
#bind-address             = 127.0.0.1

mysql -uroot -pmelodic
GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' IDENTIFIED BY 'melodic' WITH GRANT OPTION;
FLUSH PRIVILEGES;
exit

reboot system
--OR--
sudo service mysql stop
sudo service mysql start
